Энергосистема Меркурия

Code 
dataScript = FileAttachment("../data/db/data.json").json()

mercuryProduction = new Set(
  dataScript.materialProduction
    .filter(mp => mp.planet_id === "mercury")
    .map(mp => mp.material_id)
)

materialsById = new Map(dataScript.materials.map(m => [m.id, m]))
unitsById = new Map(dataScript.units.map(u => [u.id, u]))
planetsById = new Map(dataScript.planets.map(p => [p.id, p]))

// Функция для получения материалов компонента
function getComponentMaterials(componentId) {
  return dataScript.unitMaterials
    .filter(um => um.unit_id === componentId)
    .map(um => {
      const material = materialsById.get(um.material_id);
      return {
        symbol: material?.symbol,
        name: material?.name || um.material_id,
        fraction_pct: um.fraction_pct,
        isLocal: mercuryProduction.has(um.material_id)
      };
    });
}

// Функция для отображения виджета единицы
function unitWidget(unitId) {
  const unit = unitsById.get(unitId);
  if (!unit) return html`<div class="db-widget">Единица ${unitId} не найдена</div>`;

  const planet = planetsById.get(unit.production_planet_id);

  // Материалы базового юнита
  const materials = dataScript.unitMaterials
    .filter(um => um.unit_id === unitId)
    .map(um => {
      const material = materialsById.get(um.material_id);
      const isLocal = mercuryProduction.has(um.material_id);
      const massKg = unit.mass_kg ? (unit.mass_kg * um.fraction_pct / 100).toFixed(0) : null;
      return { ...um, material, isLocal, massKg };
    });

  // Компоненты (подсборки)
  const components = dataScript.unitComponents
    .filter(uc => uc.assembly_id === unitId)
    .map(uc => {
      const component = unitsById.get(uc.component_id);
      const totalMassKg = component?.mass_kg ? (component.mass_kg * uc.quantity) : null;
      const isLocal = component?.production_planet_id === unit.production_planet_id;
      const componentMaterials = getComponentMaterials(uc.component_id);

      return {
        id: uc.component_id,
        name: component?.name || uc.component_id,
        quantity: uc.quantity,
        massKg: totalMassKg,
        isLocal,
        materials: componentMaterials
      };
    });

  // Итого считаем ТОЛЬКО по материалам (не компонентам!)
  const localPct = materials.filter(m => m.isLocal).reduce((sum, m) => sum + m.fraction_pct, 0);
  const importPct = materials.filter(m => !m.isLocal).reduce((sum, m) => sum + m.fraction_pct, 0);
  const hasCriticalImports = components.some(c => !c.isLocal);

  // Строки таблицы компонентов
  const componentRows = components.map(c => html`
    <tr class="component-row">
      <td>
        <strong>${c.name}</strong>${c.quantity > 1 ? html` ×${c.quantity}` : ''}
        <div class="component-materials">
          ${c.materials.map(m => html`<span class="${m.isLocal ? 'local' : 'import'}">${m.symbol || m.name} ${m.fraction_pct}%</span>`)}
        </div>
      </td>
      <td>${c.massKg ? c.massKg.toFixed(0) : '—'} кг</td>
      <td class="${c.isLocal ? 'local' : 'import'}">${c.isLocal ? '✓' : '⚠ импорт'}</td>
    </tr>
  `);

  // Строки таблицы материалов
  const materialRows = materials.map(m => html`
    <tr>
      <td>
        ${m.material?.symbol ? html`<strong>${m.material.symbol}</strong> ` : ''}
        ${m.material?.name || m.material_id}
      </td>
      <td>${m.fraction_pct}%</td>
      <td>${m.massKg || '—'} кг</td>
      <td class="${m.isLocal ? 'local' : 'import'}">${m.isLocal ? '✓ местный' : '⚠ импорт'}</td>
    </tr>
  `);

  return html`
    <div class="db-widget">
      <h4>${unit.name} <span style="color: #6c757d; font-weight: normal;">(${unit.id})</span></h4>
      <div class="meta">
        ${unit.mass_kg ? html`<strong>${unit.mass_kg.toLocaleString()}</strong> кг` : ''}
        ${unit.power_kw ? html` | <strong>${unit.power_kw.toLocaleString()}</strong> кВт` : ''}
        ${planet ? html` | Производство: ${planet.name}` : ''}
      </div>

      ${components.length > 0 ? html`
        <div class="section-header">Компоненты</div>
        <table class="components-table">
          <thead><tr><th>Компонент</th><th>Масса</th><th>Источник</th></tr></thead>
          <tbody>${componentRows}</tbody>
        </table>
      ` : ''}

      ${materials.length > 0 ? html`
        <div class="section-header">Материалы</div>
        <table class="materials-table">
          <thead><tr><th>Материал</th><th>%</th><th>Масса</th><th>Источник</th></tr></thead>
          <tbody>${materialRows}</tbody>
        </table>
        <div class="summary">
          Итого: <span class="local-pct">${localPct.toFixed(0)}% местное</span> |
          <span class="import-pct">${importPct.toFixed(0)}% импорт</span>
          ${hasCriticalImports ? html`<br/><strong>⚠ Содержит импортные компоненты</strong>` : ''}
        </div>
      ` : ''}

      <div class="link">
        <a href="data/nomenclature.html?unit=${unitId}">Подробнее →</a>
      </div>
    </div>
  `;
}

TL;DR

Энергосистема Меркурия состоит из двух независимых подсистем:

Раздел 1: Энергия для производства - Solar Boost: 9-14 кВт/м² (vs 1.3 кВт/м² на Земле) - Power Towers: GaAs ячейки с NaK охлаждением - Крио-ЛЭП: Алюминий при -150°C — сопротивление в 10× ниже - Кремниевые панели: Алюмотермия (100% местное производство)

Раздел 2: Энергия для масс-драйверов - Пиковая мощность: 400 МВт во время запуска - Средняя мощность: 33 МВт на 1 МД (600 запусков/сутки) - Конденсаторы: Накопление энергии между запусками - Охлаждение: 40 МВт тепла через NaK + радиаторы

Обзор энергосистемы

Энергетическая инфраструктура Меркурия решает две принципиально разные задачи:

  1. Непрерывное энергоснабжение производства — заводы, печи, конвейеры (50-55 МВт на комплекс)
  2. Импульсное питание масс-драйверов — пиковые нагрузки 400 МВт, средние 33 МВт

Системы разделены, так как имеют разные режимы работы и требования.

Раздел 1: Энергия для производства

Генерация. Без этого этапа завод — груда металла.

Физика Меркурия (Solar Boost)

Локация Солнечный поток
Орбита Земли 1.3 кВт/м²
Меркурий 9–14 кВт/м²

Меркурий получает в 7-11 раз больше солнечной энергии, чем Земля. Это фундаментальное преимущество для солнечной энергетики.

Зеркала-концентраторы

Все зеркала для первой фабрики доставлены с Земли в сложенном виде.

Параметр Значение
Материал Сверхлёгкая полимерная плёнка (Каптон) с алюминиевым напылением
Вес 50 г/м²
Общий вес комплекта < 1 тонны
Конструкция Разворачиваются на углепластиковых спицах (как зонтики)
Назначение Выработка тока (на башнях) + плавка металла (в печах)

Гелио-Башни (Power Towers)

Устанавливаются роботами-альпинистами на «Пиках вечного света» (края кратера).

Принцип: Зеркала фокусируют свет на фотоэлементах.

Фотоячейки: - Тип: многопереходные на арсениде галлия (GaAs) - Свойства: дорогие, но живучие

Охлаждение (радиационное): - В вакууме конвекции нет - Грунт использовать нельзя (изолятор) - Внутри мачты циркулирует жидкий NaK, забирая тепло от раскалённых ячеек - Жидкость течёт на теневую сторону башни в радиатор - Радиатор «светит» в сторону холодного космоса (3K) или тёмного дна кратера (−150°C), сбрасывая тепло излучением

Криогенная ЛЭП

Энергию нужно спустить с горы (где свет и +400°C) вниз, в кратер (где завод и −150°C).

  • Кабель прокладывается по теневой стороне склона
  • При −150°C сопротивление алюминия падает в ~10 раз (эффект криорезистивности)
  • Потери на передачу: 3-5% вместо 15-20% при комнатной температуре
  • Можно использовать кабели меньшего сечения при той же пропускной способности

Примечание: Алюминий не является сверхпроводником ни при какой температуре. Но криогенные температуры Меркурия (~120 K в тени) снижают омическое сопротивление достаточно для эффективной передачи энергии на расстояние 1-2 км.

Производство солнечных панелей

Энергетическая независимость. Местное производство фотоэлементов.

Первый завод использует GaAs ячейки с Земли. Но для масштабирования нужно местное производство кремниевых панелей.

Почему кремний, а не GaAs

Параметр GaAs Кремний
КПД 30-40% 18-22%
Сырьё Галлий + Мышьяк (дефицит) 20% коры Меркурия
Производство Эпитаксия, чистые комнаты Проще, технологии 90-х
Вердикт Первый завод Масштабирование

Энергопотребление 1 комплекса

Потребитель Мощность Примечание
Масс-драйвер (средняя) 33 МВт 600 запусков/день × 4.8 ГДж
Солнечные печи (плавка) 10 МВт Концентраторы
Производство зеркал 5 МВт Прокатка, сварка, упаковка
Роботы (~50 шт × 5 кВт) 0.3 МВт Батарейные
Системы управления и связь 2 МВт ИТ-инфраструктура
ИТОГО ~50 МВт С запасом округляем до 55 МВт

Примечание: Расчёт по Масс-драйверам. Значение 55 МВт включает запас 10% на пиковые нагрузки и потери.

Параметры кремниевых панелей

Параметр Значение
Солнечный поток на Меркурии 10 кВт/м² (среднее по орбите)
КПД кремниевых ячеек 18%
Выход с 1 м² 10 × 0.18 = 1.8 кВт/м²
Площадь на 1 комплекс (55 МВт) 35 000 м² (квадрат 187×187 м)

Алюмотермическое восстановление кремния

Проблема: Традиционный метод (SiO₂ + C → Si + CO₂) требует углерода, которого на Меркурии нет.

Решение: Алюмотермия — восстановление алюминием.

3SiO₂ + 4Al → 3Si + 2Al₂O₃
Преимущество Описание
Алюминий местный Из шлака электролиза MRE
Нет импорта углерода Полная автономия
Побочный продукт Al₂O₃ (корунд) → керамика, абразивы

Стехиометрия: - На 1 кг кремния: 1.3 кг алюминия - Побочно: 1.3 кг корунда (Al₂O₃)

Производственная цепочка

Этап Процесс Материалы Примечание
1 Кварцевый песок (SiO₂) Местный 20% коры Меркурия
2 Алюминий Шлак MRE Уже производится
3 Алюмотермия Местное Солнечная печь, 2000°C
4 Зонная плавка — Очистка до solar grade
5 Метод Чохральского Тигель кварцевый Слиток ∅150 мм
6 Проволочная резка Алмазная проволока (Земля) Пластины 200 мкм
7 Диффузия P/B Фосфор + Бор (Земля) p-n переход
8 Трафаретная печать Алюминий местный Контакты
9 Напыление SiNx Азот (Земля) + Силан Антиотражение
10 Сборка модуля Алюминий местный Стекло не нужно (вакуум)

Энергетический баланс

3 рейса с GaAs панелями — энергетическая независимость с первого дня:

Параметр Значение Примечание
Потребление комплекса 55 МВт Завод + МД + роботы
GaAs панели (1 рейс) 35 000 м² = 56 МВт 1.75 т
GaAs панели (2 рейс) +35 000 м² = +56 МВт +1.75 т
GaAs панели (3 рейс) +35 000 м² = +56 МВт +1.75 т
Итого панелей 105 000 м² = 168 МВт 5.25 т
Запас энергии 113 МВт 3× потребление

Преимущества 3 рейсов: - Bootstrap без ожидания энергии с Роя - Возможность работы 2-3 заводов параллельно (или 1 завод + ускоренная добыча) - Резерв на пиковые нагрузки и масштабирование - Снижение критичности «первого зеркала» — можно работать сколь угодно долго на солнечной энергии

Масса панелей: 50 г/м² × 105 000 м² = 5.25 т — это ~8% от общей массы экспедиции.

Раздел 2: Энергия для масс-драйверов

Импульсная энергетика. Высокие пиковые нагрузки при запуске зеркал.

Энергопотребление масс-драйверов

На один запуск

Для запуска зеркала массой 116 кг на скорость 5 км/с требуется:

Кинетическая энергия: 1.45 ГДж
С учётом КПД 40%: 3.6 ГДж
С потерями системы: ~4.8 ГДж на запуск

Средняя мощность (1 масс-драйвер)

600 запусков/сутки × 4.8 ГДж = 2880 ГДж/сутки
= 120 ГДж/час
= 33 МВт (средняя мощность)

Пиковая мощность

Во время разгона (10 сек): 400 МВт
Между запусками: накопление в конденсаторах

Вывод: Один масс-драйвер потребляет ~33 МВт постоянно. Это эквивалент ~2-3 зеркал (через PV 18%).

Конденсаторные батареи

Проблема: Пиковая мощность 400 МВт за 10 секунд невозможна напрямую от солнечных панелей.

Решение: Конденсаторы накапливают энергию между запусками (2-3 минуты) и отдают её импульсом.

Параметр Значение
Энергия на запуск 4.8 ГДж
Время разгона 10 сек
Пиковая мощность 400 МВт
Время накопления 2-3 минуты между запусками
Средняя мощность заряда 33 МВт

Принцип работы: 1. Солнечные панели непрерывно заряжают конденсаторы (33 МВт) 2. Между запусками проходит 2-3 минуты — достаточно для накопления 4.8 ГДж 3. При запуске конденсаторы мгновенно отдают 400 МВт в катушки МД 4. Цикл повторяется 600 раз в сутки

Материалы конденсаторов: - Диэлектрик: местные силикаты + керамика - Обкладки: алюминий (местный) - Масса батареи: ~20 тонн на 1 МД

Распределение мощности по катушкам

Масс-драйвер имеет несколько сотен катушек вдоль 3-километрового трека. Энергия распределяется последовательно:

Секция Длина Катушек Мощность
Разгон (начало) 0-1 км ~100 150 МВт
Разгон (середина) 1-2 км ~100 150 МВт
Финальный разгон 2-3 км ~100 100 МВт

Управление: Катушки активируются последовательно по мере прохождения снаряда, создавая «бегущее магнитное поле».

Охлаждение систем МД (40 МВт тепла)

Проблема тепловыделения

Параметр Значение
Энергия на запуск 4.7 ГДж
КПД 40%
Тепло за выстрел 2.8 ГДж = 280 МВт за 10 сек
Интервал между выстрелами 2–3 минуты
Непрерывное тепловыделение ~40 МВт (при 600 выстрелах/сутки)

Последствия перегрева: - Перегрев катушек → падение КПД → ещё больше тепла - Тепловое расширение рельсов → заклинивание снаряда - Деградация изоляции обмоток

Решение: NaK-охлаждение + радиаторы

Система охлаждения:

Компонент Описание
Теплоноситель Эвтектический сплав NaK (натрий-калий)
Температура плавления −12°C (жидкий при любых условиях Меркурия)
Циркуляция Насосы внутри каждой катушки
Радиаторы Направлены в теневую сторону Меркурия (−180°C)
Площадь радиаторов ~5000 м² на 1 МД

Принцип работы: 1. NaK циркулирует через катушки, забирая тепло 2. Горячий NaK (~200°C) поступает в радиаторы 3. Радиаторы излучают тепло в холодный космос 4. Охлаждённый NaK (~50°C) возвращается к катушкам

Преимущество криогенных температур: - Теневая сторона Меркурия: −180°C - Разница температур ΔT = 200 − (−180) = 380°C - Высокая эффективность радиационного охлаждения

Криоалюминий

Проблема: Меди на Меркурии нет, а алюминий при комнатной температуре имеет в 1.6× худшую проводимость.

Решение: Использовать криогенное охлаждение для повышения проводимости алюминия.

Материал При +20°C При −180°C
Медь 100% 400%
Алюминий 60% 360%

Выводы: - При −180°C алюминий почти догоняет медь по проводимости - КПД масс-драйвера восстанавливается с 30% до 35-40% - Охлаждение решает сразу две задачи: отвод тепла + повышение КПД

unitWidget("CMP-103")

Code 
unitWidget("CMP-104")

Система охлаждения: - Масса: 75 тонн - Радиаторы Al: 60 т - NaK теплоноситель: 15 т - Насосы и трубопроводы: ~1 т
